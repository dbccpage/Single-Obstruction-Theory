pub fn cross_magnitude(v1: [f64; 3], v2: [f64; 3]) -> f64 {
    // Calculate the components of the cross product vector v = v1 x v2.
    // v_x = v1_y * v2_z - v1_z * v2_y
    let cross_x = v1[1] * v2[2] - v1[2] * v2[1];
    // v_y = v1_z * v2_x - v1_x * v2_z

    // magnitude = sqrt(v_x^2 + v_y^2 + v_z^2)

///
/// The magnitude of the cross product of two vectors `v1` and `v2` is equivalent to the
/// area of the parallelogram that the vectors span.
///
/// The formula for the cross product `c = v1 x v2` is:
/// `c_x = v1_y * v2_z - v1_z * v2_y`
/// `c_y = v1_z * v2_x - v1_x * v2_z`
/// `c_z = v1_x * v2_y - v1_y * v2_x`
///
/// The magnitude of the resulting vector `c` is `sqrt(c_x^2 + c_y^2 + c_z^2)`.
///
/// # Arguments
///
/// * `v1` - A 3-element array of `f64` representing the first vector `[x, y, z]`.
/// * `v2` - A 3-element array of `f64` representing the second vector `[x, y, z]`.
///
/// # Returns
///
/// A `f64` scalar representing the magnitude of the cross product.
///
/// # Examples
///
/// ```
/// // Example with orthogonal unit vectors (i-hat and j-hat)
/// let v1 = [1.0, 0.0, 0.0];
/// let v2 = [0.0, 1.0, 0.0];
/// // The cross product is k-hat [0, 0, 1], and its magnitude is 1.
/// let magnitude = cross_magnitude(v1, v2);
/// assert!((magnitude - 1.0).abs() < 1e-9);
/// ```
///
/// ```
/// // Example with parallel vectors
/// let v1 = [1.0, 2.0, 3.0];
/// let v2 = [2.0, 4.0, 6.0];
/// // The cross product of parallel vectors is the zero vector, with magnitude 0.
/// let magnitude = cross_magnitude(v1, v2);
/// assert!(magnitude.abs() < 1e-9);
/// ```
pub fn cross_magnitude(v1: [f64; 3], v2: [f64; 3]) -> f64 {
    // Calculate the components of the cross product vector
    let cross_x = v1[1] * v2[2] - v1[2] * v2[1];
    let cross_y = v1[2] * v2[0] - v1[0] * v2[2];
    let cross_z = v1[0] * v2[1] - v1[1] * v2[0];

    // Calculate the squared magnitude (sum of the squares of the components)
    let mag_sq = cross_x.powi(2) + cross_y.powi(2) + cross_z.powi(2);

    // Return the square root of the squared magnitude, which is the magnitude
    mag_sq.sqrt()
}
