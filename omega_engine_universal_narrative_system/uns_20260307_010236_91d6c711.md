pub fn attention_gradient(q: &Vec<f64>, k: &Vec<f64>, v: &Vec<f64>) -> Vec<f64> {
    let mut result = Vec::new();
    for i in 0..q.len() {
        let score = scaled_dot_product(q[i], k, v);
        let grad = (score * k[i]) - q[i] * v[i];
        result.push(grad);
    }
    result
}

/// Calculate the scaled dot-product attention
fn scaled_dot_product(q: f64, k: &Vec<f64>, v: &Vec<f64>) -> f64 {
    let scores = k.iter().zip(v.iter()).map(|(k, v)| k * v).sum::<f64>();
    scores * q
}
