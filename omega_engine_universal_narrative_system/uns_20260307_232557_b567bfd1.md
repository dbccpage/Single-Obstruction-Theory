#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash, PartialOrd, Ord)]

    /// Spatial coordinate.
    pub x: i32,

}

/// Defines the structural geometry of the space-time lattice.
/// It uses periodic boundary conditions.
#[derive(Debug, Clone)]
pub struct Lattice {
    /// Spatial extent of the lattice.
    pub l: i32,
    /// Temporal extent of the lattice.
    pub t: i32,
    /// Pre-computed list of all sites on the lattice.
    sites: Vec<Site>,
    /// Pre-computed list of all unique interaction bonds.
    bonds: Vec<(Site, Site)>,
}

impl Lattice {
    /// Creates a new lattice with given spatial and temporal dimensions.
    pub fn new(l: i32, t: i32) -> Self {
        let mut sites = Vec::with_capacity((l * t) as usize);
        for time in 0..t {
            for space in 0..l {
                sites.push(Site { x: space, t: time });
            }
        }

        let mut bonds = HashSet::new();
        for &site in &sites {
            // Spatial neighbor
            let neighbor_x = Site {
                x: (site.x + 1).rem_euclid(l),
                t: site.t,
            };
            // Temporal neighbor
            let neighbor_t = Site {
                x: site.x,
                t: (site.t + 1).rem_euclid(t),
            };

            let bond1 = if site < neighbor_x { (site, neighbor_x) } else { (neighbor_x, site) };
            let bond2 = if site < neighbor_t { (site, neighbor_t) } else { (neighbor_t, site) };
            bonds.insert(bond1);
            bonds.insert(bond2);
        }

        Lattice {
            l,
            t,
            sites,
            bonds: bonds.into_iter().collect(),
        }
    }
}
