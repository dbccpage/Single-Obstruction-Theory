pub fn calculate_refraction_angle(n1: f64, theta1_rad: f64, n2: f64) -> Result<f64, &'static str> {
    // Constraint: The indices of refraction n1 and n2 must be strictly greater than 0.
    if n1 <= 0.0 || n2 <= 0.0 {
        return Err("Indices of refraction must be positive.");
    }

    // From Snell's Law: sin(theta_2) = (n1 / n2) * sin(theta_1)
    let sin_theta2_arg = (n1 / n2) * theta1_rad.sin();

    // The domain of arcsin is [-1, 1]. If the argument is outside this range,
    // it corresponds to total internal reflection.
    if sin_theta2_arg > 1.0 || sin_theta2_arg < -1.0 {
        return Err("Total internal reflection.");
    }

    // Calculate theta_2 by taking the arcsin. This returns the principal value.
    let theta2_rad = sin_theta2_arg.asin();

    Ok(theta2_rad)
}
